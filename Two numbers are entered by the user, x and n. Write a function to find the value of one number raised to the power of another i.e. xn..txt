import java.util.*;
public class Main {
    public static double calculatePower(double x, double n){

        return Math.pow(x,n);
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter base number (x) : ");
        double x = sc.nextDouble();
        System.out.println("Enter exponent (n) : ");
        double n = sc.nextDouble();
        System.out.println(n + " The raise to the Power of " + x +" is : "+ calculatePower(x,n));

        }
    }
